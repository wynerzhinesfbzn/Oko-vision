🧩 Шаг 2. Создание контекста кошелька (управление аккаунтом)

Создаём src/contexts/WalletContext/WalletContext.tsx:

```tsx
import React, { createContext, useContext, useState, useEffect } from 'react';
import { ethers } from 'ethers';

interface WalletContextType {
  address: string | null;
  balance: string | null;
  mnemonic: string | null;
  createNewWallet: () => void;
  importWallet: (mnemonic: string) => void;
  connectToRobinhood: () => Promise<void>;
  disconnectWallet: () => void;
}

const WalletContext = createContext<WalletContextType | null>(null);

export const WalletProvider = ({ children }: { children: React.ReactNode }) => {
  const [address, setAddress] = useState<string | null>(null);
  const [balance, setBalance] = useState<string | null>(null);
  const [mnemonic, setMnemonic] = useState<string | null>(null);

  // Генерация нового кошелька
  const createNewWallet = () => {
    const wallet = ethers.Wallet.createRandom();
    setMnemonic(wallet.mnemonic?.phrase || null);
    setAddress(wallet.address);
    // Сохраняем в localStorage для сессии
    localStorage.setItem('firewallet_address', wallet.address);
    localStorage.setItem('firewallet_mnemonic', wallet.mnemonic?.phrase || '');
  };

  // Импорт по мнемонике
  const importWallet = (phrase: string) => {
    try {
      const wallet = ethers.Wallet.fromMnemonic(phrase);
      setAddress(wallet.address);
      setMnemonic(phrase);
      localStorage.setItem('firewallet_address', wallet.address);
      localStorage.setItem('firewallet_mnemonic', phrase);
    } catch (e) {
      alert('Неверная мнемоника');
    }
  };

  // Подключение к сети Robinhood Chain (EVM)
  const connectToRobinhood = async () => {
    // Здесь можно использовать window.ethereum для MetaMask, но если у нас уже есть адрес – просто показываем
    if (!address) {
      alert('Сначала создайте или импортируйте кошелёк');
      return;
    }
    // Получаем баланс через публичный RPC Robinhood
    const provider = new ethers.JsonRpcProvider('https://rpc.mainnet.chain.robinhood.com');
    const balanceWei = await provider.getBalance(address);
    setBalance(ethers.formatEther(balanceWei));
  };

  const disconnectWallet = () => {
    setAddress(null);
    setBalance(null);
    setMnemonic(null);
    localStorage.removeItem('firewallet_address');
    localStorage.removeItem('firewallet_mnemonic');
  };

  // При загрузке проверяем сохранённый кошелёк
  useEffect(() => {
    const savedAddress = localStorage.getItem('firewallet_address');
    const savedMnemonic = localStorage.getItem('firewallet_mnemonic');
    if (savedAddress && savedMnemonic) {
      setAddress(savedAddress);
      setMnemonic(savedMnemonic);
      // автоматически подключаемся к сети
      connectToRobinhood();
    }
  }, []);

  return (
    <WalletContext.Provider value={{
      address,
      balance,
      mnemonic,
      createNewWallet,
      importWallet,
      connectToRobinhood,
      disconnectWallet
    }}>
      {children}
    </WalletContext.Provider>
  );
};

export const useWallet = () => {
  const ctx = useContext(WalletContext);
  if (!ctx) throw new Error('useWallet must be used within WalletProvider');
  return ctx;
};
```

🎨 Шаг 3. Дизайн входа/создания кошелька (Robinhood стиль)

Заменим старую страницу входа на экран приветствия, где пользователь может создать новый кошелёк или импортировать существующий (как в Robinhood).

src/pages/Onboarding/Onboarding.tsx

```tsx
import React, { useState } from 'react';
import { useWallet } from '../../contexts/WalletContext/WalletContext';
import { motion } from 'framer-motion';

const Onboarding = () => {
  const { createNewWallet, importWallet } = useWallet();
  const [mnemonicInput, setMnemonicInput] = useState('');
  const [showImport, setShowImport] = useState(false);

  return (
    <div className="min-h-screen bg-[#0a0a0a] flex items-center justify-center p-4">
      <motion.div
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        className="w-full max-w-md bg-[#1a1a1a] rounded-3xl p-8 border border-white/10 shadow-2xl"
      >
        <div className="text-center mb-8">
          <div className="text-4xl font-light text-white">Fire Wallet</div>
          <div className="text-gray-400 text-sm mt-1">by Robinhood Chain</div>
        </div>

        {!showImport ? (
          <>
            <button
              onClick={createNewWallet}
              className="w-full bg-[#2962ff] hover:bg-blue-700 text-white font-medium py-3 rounded-xl transition-all mb-3"
            >
              Создать новый кошелёк
            </button>
            <button
              onClick={() => setShowImport(true)}
              className="w-full bg-white/5 hover:bg-white/10 text-white py-3 rounded-xl transition-all border border-white/10"
            >
              Импортировать кошелёк
            </button>
          </>
        ) : (
          <>
            <textarea
              value={mnemonicInput}
              onChange={(e) => setMnemonicInput(e.target.value)}
              placeholder="Введите 12 или 24 слова через пробел"
              className="w-full bg-white/10 border border-white/20 rounded-xl px-4 py-3 text-white placeholder-gray-400 focus:outline-none focus:ring-2 focus:ring-blue-500 h-24 mb-4"
            />
            <button
              onClick={() => importWallet(mnemonicInput)}
              className="w-full bg-[#2962ff] hover:bg-blue-700 text-white font-medium py-3 rounded-xl transition-all"
            >
              Импортировать
            </button>
            <button
              onClick={() => setShowImport(false)}
              className="w-full text-gray-400 text-sm mt-3 hover:text-white"
            >
              Назад
            </button>
          </>
        )}

        <div className="mt-6 text-center text-xs text-gray-500">
          🔒 Защищено мнемонической фразой
        </div>
      </motion.div>
    </div>
  );
};

export default Onboarding;
```

🧩 Шаг 5. Интеграция в App.tsx

```tsx
import { AuthProvider, useAuth } from './contexts/AuthContext/AuthContext';
import { WalletProvider } from './contexts/WalletContext/WalletContext';
import Onboarding from './pages/Onboarding/Onboarding';
import Dashboard from './pages/Dashboard/Dashboard';

const AppContent = () => {
  const { isAuthenticated } = useAuth();
  const { address } = useWallet();

  // Если не авторизован (биометрия/пароль) – показываем Onboarding (создание кошелька)
  if (!isAuthenticated) {
    return <Onboarding />;
  }

  // Если авторизован, но нет кошелька – тоже Onboarding (но уже создание)
  if (!address) {
    return <Onboarding />;
  }

  return <Dashboard />;
};

function App() {
  return (
    <AuthProvider>
      <WalletProvider>
        <AppContent />
      </WalletProvider>
    </AuthProvider>
  );
}

export default App;
```
