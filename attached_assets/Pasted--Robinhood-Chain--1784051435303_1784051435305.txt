📦 Шаг 1. Установка зависимостей

В корне проекта (где package.json) выполни:

🧩 Шаг 2. Создаём контекст авторизации

Создай папку src/contexts/AuthContext/ и внутри файлы:

src/contexts/AuthContext/types.ts

```typescript
export type AuthMethod = 'password' | 'biometrics' | 'mnemonic' | 'crypto' | 'none';

export interface AuthState {
  isAuthenticated: boolean;
  method: AuthMethod | null;
  user: { name?: string; email?: string } | null;
}
```

src/contexts/AuthContext/AuthContext.tsx

```tsx
import React, { createContext, useContext, useState, useEffect, ReactNode } from 'react';
import { AuthState, AuthMethod } from './types';

interface AuthContextType extends AuthState {
  login: (method: AuthMethod, credentials: any) => Promise<boolean>;
  logout: () => void;
  setBiometricsSupported: (supported: boolean) => void;
  biometricsSupported: boolean;
}

const AuthContext = createContext<AuthContextType | undefined>(undefined);

export const AuthProvider = ({ children }: { children: ReactNode }) => {
  const [state, setState] = useState<AuthState>({
    isAuthenticated: false,
    method: null,
    user: null,
  });
  const [biometricsSupported, setBiometricsSupported] = useState(false);

  // Проверка поддержки WebAuthn при монтировании
  useEffect(() => {
    const checkBiometrics = async () => {
      if (window.PublicKeyCredential) {
        try {
          const available = await PublicKeyCredential.isUserVerifyingPlatformAuthenticatorAvailable();
          setBiometricsSupported(available);
        } catch {
          setBiometricsSupported(false);
        }
      } else {
        setBiometricsSupported(false);
      }
    };
    checkBiometrics();
  }, []);

  // Метод входа
  const login = async (method: AuthMethod, credentials: any): Promise<boolean> => {
    // Здесь будет логика проверки
    // Сейчас просто имитация
    if (method === 'password') {
      // проверка пароля (позже реализуем)
      if (credentials.password === '123456') {
        setState({ isAuthenticated: true, method, user: { name: 'User' } });
        return true;
      }
      return false;
    }
    if (method === 'biometrics') {
      // WebAuthn вызов
      const success = await authenticateWithBiometrics();
      if (success) {
        setState({ isAuthenticated: true, method, user: { name: 'User' } });
        return true;
      }
      return false;
    }
    if (method === 'mnemonic') {
      // проверка мнемонической фразы
      if (credentials.mnemonic === 'word1 word2 word3 word4 word5 word6') {
        setState({ isAuthenticated: true, method, user: { name: 'User' } });
        return true;
      }
      return false;
    }
    // crypto - возможно, аппаратный ключ или криптографическая подпись
    if (method === 'crypto') {
      // имитация
      setState({ isAuthenticated: true, method, user: { name: 'User' } });
      return true;
    }
    return false;
  };

  const logout = () => {
    setState({ isAuthenticated: false, method: null, user: null });
  };

  const authenticateWithBiometrics = async (): Promise<boolean> => {
    // Здесь будет реальный вызов WebAuthn (позже)
    // Сейчас имитация успеха
    return true;
  };

  return (
    <AuthContext.Provider
      value={{
        ...state,
        login,
        logout,
        biometricsSupported,
        setBiometricsSupported,
      }}
    >
      {children}
    </AuthContext.Provider>
  );
};

export const useAuth = () => {
  const context = useContext(AuthContext);
  if (!context) throw new Error('useAuth must be used within AuthProvider');
  return context;
};
```

🎨 Шаг 3. Создаём страницу входа (Login) в стиле Robinhood + Fire

Создай папку src/pages/Login/ и файлы:

src/pages/Login/Login.tsx

```tsx
import React, { useState } from 'react';
import { useAuth } from '../../contexts/AuthContext/AuthContext';
import { motion } from 'framer-motion';
import { FingerprintIcon, FaceSmileIcon, KeyIcon, ShieldCheckIcon } from '@heroicons/react/24/outline';

const Login: React.FC = () => {
  const { login, biometricsSupported } = useAuth();
  const [method, setMethod] = useState<'password' | 'biometrics' | 'mnemonic' | 'crypto'>('password');
  const [password, setPassword] = useState('');
  const [mnemonic, setMnemonic] = useState('');
  const [error, setError] = useState('');

  const handleLogin = async () => {
    let success = false;
    if (method === 'password') {
      success = await login('password', { password });
    } else if (method === 'biometrics') {
      success = await login('biometrics', {});
    } else if (method === 'mnemonic') {
      success = await login('mnemonic', { mnemonic });
    } else if (method === 'crypto') {
      success = await login('crypto', {});
    }
    if (!success) {
      setError('Неверные данные. Попробуйте снова.');
    } else {
      setError('');
    }
  };

  return (
    <div className="min-h-screen flex items-center justify-center bg-gradient-to-br from-gray-900 via-black to-orange-900/30 p-4">
      <motion.div
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        transition={{ duration: 0.6 }}
        className="w-full max-w-md bg-white/5 backdrop-blur-xl border border-white/10 rounded-3xl p-8 shadow-2xl"
      >
        <div className="text-center mb-8">
          <h1 className="text-4xl font-bold text-white tracking-tight">
            Fire<span className="text-orange-500">Wallet</span>
          </h1>
          <p className="text-gray-400 mt-2 text-sm">Войдите в свой огненный кошелёк</p>
        </div>

        {/* Выбор метода */}
        <div className="grid grid-cols-2 gap-3 mb-6">
          <button
            onClick={() => setMethod('password')}
            className={`flex items-center justify-center gap-2 py-3 rounded-xl border transition-all ${
              method === 'password'
                ? 'border-orange-500 bg-orange-500/20 text-orange-400'
                : 'border-white/10 bg-white/5 text-gray-400 hover:bg-white/10'
            }`}
          >
            <KeyIcon className="w-5 h-5" />
            <span>Пароль</span>
          </button>
          <button
            onClick={() => setMethod('biometrics')}
            disabled={!biometricsSupported}
            className={`flex items-center justify-center gap-2 py-3 rounded-xl border transition-all ${
              method === 'biometrics'
                ? 'border-orange-500 bg-orange-500/20 text-orange-400'
                : 'border-white/10 bg-white/5 text-gray-400 hover:bg-white/10'
            } ${!biometricsSupported && 'opacity-50 cursor-not-allowed'}`}
          >
            <FingerprintIcon className="w-5 h-5" />
            <span>Биометрия</span>
          </button>
          <button
            onClick={() => setMethod('mnemonic')}
            className={`flex items-center justify-center gap-2 py-3 rounded-xl border transition-all ${
              method === 'mnemonic'
                ? 'border-orange-500 bg-orange-500/20 text-orange-400'
                : 'border-white/10 bg-white/5 text-gray-400 hover:bg-white/10'
            }`}
          >
            <ShieldCheckIcon className="w-5 h-5" />
            <span>6 слов</span>
          </button>
          <button
            onClick={() => setMethod('crypto')}
            className={`flex items-center justify-center gap-2 py-3 rounded-xl border transition-all ${
              method === 'crypto'
                ? 'border-orange-500 bg-orange-500/20 text-orange-400'
                : 'border-white/10 bg-white/5 text-gray-400 hover:bg-white/10'
            }`}
          >
            <FaceSmileIcon className="w-5 h-5" />
            <span>Крипто-ключ</span>
          </button>
        </div>

        {/* Поля ввода в зависимости от метода */}
        {method === 'password' && (
          <div className="mb-6">
            <input
              type="password"
              value={password}
              onChange={(e) => setPassword(e.target.value)}
              placeholder="Введите пароль"
              className="w-full bg-white/10 border border-white/20 rounded-xl px-4 py-3 text-white placeholder-gray-400 focus:outline-none focus:ring-2 focus:ring-orange-500"
            />
          </div>
        )}
        {method === 'mnemonic' && (
          <div className="mb-6">
            <input
              type="text"
              value={mnemonic}
              onChange={(e) => setMnemonic(e.target.value)}
              placeholder="Введите 6 слов через пробел"
              className="w-full bg-white/10 border border-white/20 rounded-xl px-4 py-3 text-white placeholder-gray-400 focus:outline-none focus:ring-2 focus:ring-orange-500"
            />
          </div>
        )}
        {method === 'biometrics' && (
          <div className="mb-6 text-center text-gray-300">
            <FingerprintIcon className="w-16 h-16 mx-auto text-orange-400 mb-2" />
            <p>Нажмите кнопку ниже и подтвердите свою личность</p>
          </div>
        )}
        {method === 'crypto' && (
          <div className="mb-6 text-center text-gray-300">
            <ShieldCheckIcon className="w-16 h-16 mx-auto text-orange-400 mb-2" />
            <p>Подключите аппаратный ключ (YubiKey и т.п.)</p>
          </div>
        )}

        {error && <p className="text-red-400 text-sm mb-4 text-center">{error}</p>}

        <button
          onClick={handleLogin}
          className="w-full bg-gradient-to-r from-orange-500 to-amber-500 text-white font-semibold py-3 rounded-xl hover:opacity-90 transition-all shadow-lg shadow-orange-500/30"
        >
          Войти в кошелёк
        </button>

        <div className="mt-6 text-center text-xs text-gray-500">
          Защищено огнём 🔥
        </div>
      </motion.div>
    </div>
  );
};

export default Login;
```

🧩 Шаг 4. Интеграция в приложение

Обнови корневой App.tsx или main.tsx:

```tsx
// В main.tsx оберни приложение в AuthProvider
import { AuthProvider } from './contexts/AuthContext/AuthContext';
import Login from './pages/Login/Login';
import { useAuth } from './contexts/AuthContext/AuthContext';

const AppContent = () => {
  const { isAuthenticated } = useAuth();
  return isAuthenticated ? <YourExistingApp /> : <Login />;
};

// В корневом рендере:
<AuthProvider>
  <AppContent />
</AuthProvider>
```
