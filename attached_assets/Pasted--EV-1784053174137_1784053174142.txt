```typescript
import { ethers } from 'ethers';

// РЕАЛЬНОЕ создание кошелька
export const createRealWallet = () => {
  const wallet = ethers.Wallet.createRandom();
  return {
    address: wallet.address,
    mnemonic: wallet.mnemonic?.phrase || '',
    privateKey: wallet.privateKey,
  };
};

// РЕАЛЬНЫЙ импорт из мнемоники
export const importRealWallet = (mnemonic: string) => {
  try {
    const wallet = ethers.Wallet.fromMnemonic(mnemonic);
    return {
      address: wallet.address,
      mnemonic: mnemonic,
      privateKey: wallet.privateKey,
    };
  } catch (e) {
    throw new Error('Неверная мнемоническая фраза');
  }
};

// РЕАЛЬНЫЙ баланс через RPC
export const getRealBalance = async (address: string) => {
  try {
    const provider = new ethers.JsonRpcProvider('https://rpc.mainnet.chain.robinhood.com');
    const balance = await provider.getBalance(address);
    return ethers.formatEther(balance);
  } catch (e) {
    console.error('Ошибка получения баланса:', e);
    return '0';
  }
};

// РЕАЛЬНАЯ отправка ETH
export const sendRealTransaction = async (privateKey: string, to: string, amount: string) => {
  try {
    const provider = new ethers.JsonRpcProvider('https://rpc.mainnet.chain.robinhood.com');
    const wallet = new ethers.Wallet(privateKey, provider);
    const tx = await wallet.sendTransaction({
      to: to,
      value: ethers.parseEther(amount),
    });
    await tx.wait();
    return tx.hash;
  } catch (e) {
    throw new Error('Ошибка отправки: ' + (e as Error).message);
  }
};
```

```tsx
import React, { createContext, useContext, useState, useEffect } from 'react';
import { createRealWallet, importRealWallet, getRealBalance, sendRealTransaction } from '../../utils/wallet';

interface WalletContextType {
  address: string | null;
  balance: string;
  privateKey: string | null;
  mnemonic: string | null;
  createNewWallet: () => void;
  importWallet: (mnemonic: string) => void;
  refreshBalance: () => Promise<void>;
  sendTransaction: (to: string, amount: string) => Promise<string>;
  disconnectWallet: () => void;
}

const WalletContext = createContext<WalletContextType | null>(null);

export const WalletProvider = ({ children }: { children: React.ReactNode }) => {
  const [address, setAddress] = useState<string | null>(null);
  const [balance, setBalance] = useState('0');
  const [privateKey, setPrivateKey] = useState<string | null>(null);
  const [mnemonic, setMnemonic] = useState<string | null>(null);

  const refreshBalance = async () => {
    if (address) {
      const bal = await getRealBalance(address);
      setBalance(bal);
    }
  };

  const createNewWallet = () => {
    const wallet = createRealWallet();
    setAddress(wallet.address);
    setPrivateKey(wallet.privateKey);
    setMnemonic(wallet.mnemonic);
    localStorage.setItem('firewallet_address', wallet.address);
    localStorage.setItem('firewallet_privateKey', wallet.privateKey);
    localStorage.setItem('firewallet_mnemonic', wallet.mnemonic);
    refreshBalance();
  };

  const importWallet = (phrase: string) => {
    try {
      const wallet = importRealWallet(phrase);
      setAddress(wallet.address);
      setPrivateKey(wallet.privateKey);
      setMnemonic(wallet.mnemonic);
      localStorage.setItem('firewallet_address', wallet.address);
      localStorage.setItem('firewallet_privateKey', wallet.privateKey);
      localStorage.setItem('firewallet_mnemonic', wallet.mnemonic);
      refreshBalance();
    } catch (e) {
      throw new Error('Неверная мнемоника');
    }
  };

  const sendTransaction = async (to: string, amount: string) => {
    if (!privateKey) throw new Error('Нет приватного ключа');
    const hash = await sendRealTransaction(privateKey, to, amount);
    await refreshBalance();
    return hash;
  };

  const disconnectWallet = () => {
    setAddress(null);
    setBalance('0');
    setPrivateKey(null);
    setMnemonic(null);
    localStorage.removeItem('firewallet_address');
    localStorage.removeItem('firewallet_privateKey');
    localStorage.removeItem('firewallet_mnemonic');
  };

  useEffect(() => {
    const savedAddress = localStorage.getItem('firewallet_address');
    const savedPrivateKey = localStorage.getItem('firewallet_privateKey');
    if (savedAddress && savedPrivateKey) {
      setAddress(savedAddress);
      setPrivateKey(savedPrivateKey);
      refreshBalance();
    }
  }, []);

  return (
    <WalletContext.Provider value={{
      address,
      balance,
      privateKey,
      mnemonic,
      createNewWallet,
      importWallet,
      refreshBalance,
      sendTransaction,
      disconnectWallet
    }}>
      {children}
    </WalletContext.Provider>
  );
};

export const useWallet = () => {
  const ctx = useContext(WalletContext);
  if (!ctx) throw new Error('useWallet must be used within WalletProvider');
  return ctx;
};
```

```typescript
import axios from 'axios';

const CHAIN_ID = 4663; // Robinhood Chain
const BASE_URL = 'https://api.1inch.dev/swap/v5.2';

// РЕАЛЬНЫЙ запрос курса
export const getRealQuote = async (fromToken: string, toToken: string, amount: string) => {
  try {
    const url = `${BASE_URL}/${CHAIN_ID}/quote?fromTokenAddress=${fromToken}&toTokenAddress=${toToken}&amount=${amount}`;
    const response = await axios.get(url, {
      headers: { 'Accept': 'application/json' }
    });
    return response.data;
  } catch (error) {
    console.error('Ошибка получения курса:', error);
    return null;
  }
};

// РЕАЛЬНОЕ выполнение свапа
export const getRealSwap = async (
  fromToken: string, 
  toToken: string, 
  amount: string, 
  fromAddress: string, 
  slippage: number = 1
) => {
  try {
    const url = `${BASE_URL}/${CHAIN_ID}/swap?fromTokenAddress=${fromToken}&toTokenAddress=${toToken}&amount=${amount}&fromAddress=${fromAddress}&slippage=${slippage}`;
    const response = await axios.get(url, {
      headers: { 'Accept': 'application/json' }
    });
    return response.data;
  } catch (error) {
    console.error('Ошибка создания свопа:', error);
    return null;
  }
```

```typescript
import axios from 'axios';

const LI_FI_API = 'https://li.quest/v1';

// РЕАЛЬНЫЙ запрос на мост
export const getBridgeQuote = async (
  fromChain: string,
  toChain: string,
  fromToken: string,
  toToken: string,
  fromAmount: string,
  fromAddress: string
) => {
  try {
    const url = `${LI_FI_API}/quote?fromChain=${fromChain}&toChain=${toChain}&fromToken=${fromToken}&toToken=${toToken}&fromAmount=${fromAmount}&fromAddress=${fromAddress}`;
    const response = await axios.get(url);
    return response.data;
  } catch (error) {
    console.error('Ошибка получения моста:', error);
    return null;
  }
};

// РЕАЛЬНОЕ выполнение бриджа
export const executeBridge = async (quote: any) => {
  try {
    const url = `${LI_FI_API}/execute`;
    const response = await axios.post(url, quote);
    return response.data;
  } catch (error) {
    console.error('Ошибка выполнения бриджа:', error);
    return null;
  }
};

// ID сетей для LI.FI
export const CHAIN_IDS = {
  ROBINHOOD: 4663,
  SOLANA: 501,
  ETHEREUM: 1,
  BSC: 56,
  POLYGON: 137,
  ARBITRUM: 42161,
  OPTIMISM: 10,
  BASE: 8453,
};
```

🎨 Шаг 5. Полный интерфейс с DEX и мостом

Создай src/components/Swap.tsx (полностью рабочий):

```tsx
import React, { useState } from 'react';
import { useWallet } from '../contexts/WalletContext/WalletContext';
import { getRealQuote, getRealSwap, REAL_TOKENS } from '../api/1inch';
import { getBridgeQuote, executeBridge, CHAIN_IDS } from '../api/lifi';

const Swap = () => {
  const { address, balance, privateKey, refreshBalance, sendTransaction } = useWallet();
  const [fromToken, setFromToken] = useState('WETH');
  const [toToken, setToToken] = useState('USDC');
  const [amount, setAmount] = useState('');
  const [quote, setQuote] = useState<any>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState('');
  const [bridgeMode, setBridgeMode] = useState(false);

  const fetchQuote = async () => {
    if (!amount || parseFloat(amount) <= 0) {
      setError('Введите сумму');
      return;
    }
    setLoading(true);
    setError('');
    try {
      if (bridgeMode) {
        // Получаем маршрут моста
        const result = await getBridgeQuote(
          CHAIN_IDS.ROBINHOOD,
          CHAIN_IDS.SOLANA,
          REAL_TOKENS[fromToken as keyof typeof REAL_TOKENS],
          REAL_TOKENS[toToken as keyof typeof REAL_TOKENS],
          amount,
          address || ''
        );
        setQuote(result);
      } else {
        const fromAddress = REAL_TOKENS[fromToken as keyof typeof REAL_TOKENS];
        const toAddress = REAL_TOKENS[toToken as keyof typeof REAL_TOKENS];
        const result = await getRealQuote(fromAddress, toAddress, amount);
        setQuote(result);
      }
    } catch (e) {
      setError('Ошибка запроса');
    } finally {
      setLoading(false);
    }
  };

  const executeSwap = async () => {
    if (!address) {
      setError('Создайте кошелёк');
      return;
    }
    if (!privateKey) {
      setError('Нет приватного ключа');
      return;
    }
    setLoading(true);
    setError('');
    try {
      if (bridgeMode) {
        // Выполняем бридж
        const result = await executeBridge(quote);
        if (result) {
          alert('Мост выполнен успешно! Транзакция: ' + result.txHash);
          await refreshBalance();
        }
      } else {
        const fromAddress = REAL_TOKENS[fromToken as keyof typeof REAL_TOKENS];
        const toAddress = REAL_TOKENS[toToken as keyof typeof REAL_TOKENS];
        const result = await getRealSwap(fromAddress, toAddress, amount, address);
        if (result && result.tx) {
          // Отправляем реальную транзакцию
          const txHash = await sendTransaction(
            result.tx.to,
            result.tx.value || '0'
          );
          alert('Обмен выполнен! Хэш: ' + txHash);
          await refreshBalance();
        }
      }
    } catch (e) {
      setError((e as Error).message || 'Ошибка выполнения');
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="bg-[#1a1a1a] rounded-2xl p-6 border border-white/10 max-w-md mx-auto">
      <h2 className="text-xl font-light text-white mb-4">
        {bridgeMode ? '🌉 Мост (Bridge)' : '🔄 Обмен (Swap)'}
      </h2>

      <div className="flex gap-2 mb-4">
        <button
          onClick={() => setBridgeMode(false)}
          className={`flex-1 py-2 rounded-xl text-sm font-medium transition-all ${
            !bridgeMode 
              ? 'bg-[#2962ff] text-white' 
              : 'bg-white/5 text-gray-400 hover:bg-white/10'
          }`}
        >
          Обмен
        </button>
        <button
          onClick={() => setBridgeMode(true)}
          className={`flex-1 py-2 rounded-xl text-sm font-medium transition-all ${
            bridgeMode 
              ? 'bg-[#2962ff] text-white' 
              : 'bg-white/5 text-gray-400 hover:bg-white/10'
          }`}
        >
          Мост
        </button>
      </div>
      
      <div className="space-y-3">
        <div>
          <label className="text-gray-400 text-sm">Отдаёте</label>
          <div className="flex gap-2">
            <input
              type="number"
              value={amount}
              onChange={(e) => setAmount(e.target.value)}
              placeholder="0.0"
              className="flex-1 bg-white/10 border border-white/20 rounded-xl px-4 py-3 text-white"
            />
            <select
              value={fromToken}
              onChange={(e) => setFromToken(e.target.value)}
              className="bg-white/10 border border-white/20 rounded-xl px-4 py-3 text-white"
            >
              <option value="WETH">ETH</option>
              <option value="USDC">USDC</option>
              <option value="CASHCAT">CASHCAT</option>
              <option value="STONKS">STONKS</option>
            </select>
          </div>
        </div>

        <div className="text-center text-gray-400 text-sm">⬇️</div>

        <div>
          <label className="text-gray-400 text-sm">Получаете</label>
          <div className="flex gap-2">
            <input
              type="text"
              value={quote ? (parseFloat(quote.toAmount) / 1e18).toFixed(6) : ''}
              placeholder="0.0"
              readOnly
              className="flex-1 bg-white/10 border border-white/20 rounded-xl px-4 py-3 text-white opacity-70"
            />
            <select
              value={toToken}
              onChange={(e) => setToToken(e.target.value)}
              className="bg-white/10 border border-white/20 rounded-xl px-4 py-3 text-white"
            >
              <option value="USDC">USDC</option>
              <option value="WETH">ETH</option>
              <option value="CASHCAT">CASHCAT</option>
              <option value="STONKS">STONKS</option>
            </select>
          </div>
        </div>

        {error && <div className="text-red-400 text-sm">{error}</div>}

        <div className="flex gap-2">
          <button
            onClick={fetchQuote}
            className="flex-1 bg-white/10 hover:bg-white/20 py-3 rounded-xl transition-all text-white"
            disabled={loading}
          >
            {loading ? 'Загрузка...' : 'Получить курс'}
          </button>
          <button
            onClick={executeSwap}
            className="flex-1 bg-[#2962ff] hover:bg-blue-700 py-3 rounded-xl transition-all text-white font-medium"
            disabled={!quote || loading || !address}
          >
            {bridgeMode ? 'Переместить' : 'Обменять'}
          </button>
        </div>

        {quote && (
          <div className="text-xs text-gray-400 mt-2">
            Курс: 1 {fromToken} ≈ {(parseFloat(quote.toAmount) / parseFloat(amount) / 1e18).toFixed(4)} {toToken}
            {bridgeMode && ' (через мост)'}
          </div>
        )}

        <div className="text-center text-xs text-gray-500 mt-4">
          {address ? `Адрес: ${address.slice(0, 6)}...${address.slice(-4)}` : 'Кошелёк не создан'}
          <br />
          Баланс: {parseFloat(balance).toFixed(4)} ETH
        </div>
      </div>
    </div>
  );
};

export default Swap;
```
