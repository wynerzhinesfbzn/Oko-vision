🧱 Часть 1. Аутентификация с реальной биометрией (WebAuthn) + пароль / мнемоника

Используем библиотеку @simplewebauthn/browser для работы с WebAuthn. Регистрация и логин будут работать без бэкенда – данные хранятся в localStorage (для демонстрации; в продакшене нужен сервер).

📦 Установка зависимостей

```bash
pnpm add @simplewebauthn/browser @simplewebauthn/server
```

🔐 src/contexts/AuthContext/AuthContext.tsx (полный)

```tsx
import React, { createContext, useContext, useState, useEffect } from 'react';
import { 
  startRegistration, 
  startAuthentication, 
  browserSupportsWebAuthn 
} from '@simplewebauthn/browser';
import { generateRegistrationOptions, verifyRegistrationResponse } from '@simplewebauthn/server';

// Вспомогательные функции (замена серверной логики)
const generateChallenge = () => {
  const challenge = new Uint8Array(32);
  crypto.getRandomValues(challenge);
  return challenge;
};

// Сохранение регистрации в localStorage
const saveCredential = (credential: any) => {
  const existing = JSON.parse(localStorage.getItem('webauthn_creds') || '[]');
  existing.push(credential);
  localStorage.setItem('webauthn_creds', JSON.stringify(existing));
};

const getCredentials = () => {
  return JSON.parse(localStorage.getItem('webauthn_creds') || '[]');
};

interface AuthContextType {
  isAuthenticated: boolean;
  loginWithPassword: (pass: string) => boolean;
  loginWithMnemonic: (mnemonic: string) => boolean;
  loginWithBiometrics: () => Promise<boolean>;
  registerBiometrics: () => Promise<boolean>;
  logout: () => void;
  user: { name: string } | null;
  biometricsSupported: boolean;
}

const AuthContext = createContext<AuthContextType | null>(null);

export const AuthProvider = ({ children }: { children: React.ReactNode }) => {
  const [isAuthenticated, setIsAuthenticated] = useState(false);
  const [user, setUser] = useState<{ name: string } | null>(null);
  const [biometricsSupported, setBiometricsSupported] = useState(false);

  useEffect(() => {
    setBiometricsSupported(browserSupportsWebAuthn());
    // Проверяем, есть ли сохраненная сессия
    const session = localStorage.getItem('auth_session');
    if (session) {
      setIsAuthenticated(true);
      setUser({ name: 'User' });
    }
  }, []);

  // --- Пароль (простой, для примера) ---
  const loginWithPassword = (pass: string) => {
    if (pass === '123456') { // замени на реальный хеш
      localStorage.setItem('auth_session', 'true');
      setIsAuthenticated(true);
      setUser({ name: 'User' });
      return true;
    }
    return false;
  };

  // --- Мнемоника (6 слов) ---
  const loginWithMnemonic = (mnemonic: string) => {
    if (mnemonic === 'word1 word2 word3 word4 word5 word6') {
      localStorage.setItem('auth_session', 'true');
      setIsAuthenticated(true);
      setUser({ name: 'User' });
      return true;
    }
    return false;
  };

  // --- WebAuthn: регистрация ---
  const registerBiometrics = async (): Promise<boolean> => {
    try {
      const challenge = generateChallenge();
      const options = {
        rp: { name: 'Fire Wallet' },
        user: { id: 'user-id', name: 'user@example.com', displayName: 'User' },
        challenge,
        authenticatorSelection: { userVerification: 'required' },
      };
      const attResp = await startRegistration(options);
      // Валидация (упрощенно, без сервера)
      const verification = await verifyRegistrationResponse({
        response: attResp,
        expectedChallenge: challenge.toString(),
        expectedOrigin: window.location.origin,
        expectedRPID: window.location.hostname,
      });
      if (verification.verified) {
        saveCredential(attResp);
        return true;
      }
      return false;
    } catch (error) {
      console.error(error);
      return false;
    }
  };

  // --- WebAuthn: логин ---
  const loginWithBiometrics = async (): Promise<boolean> => {
    try {
      const creds = getCredentials();
      if (creds.length === 0) {
        // Нет зарегистрированных устройств – предлагаем зарегистрироваться
        alert('Сначала зарегистрируйте биометрию через настройки');
        return false;
      }
      // Берем первый сохраненный credential (для упрощения)
      const credential = creds[0];
      const challenge = generateChallenge();
      const options = {
        challenge,
        allowCredentials: [{ id: credential.id, type: 'public-key' }],
        userVerification: 'required',
      };
      const authResp = await startAuthentication(options);
      // Валидация (упрощенно)
      // Здесь должна быть проверка подписи, но для демо просто проверяем наличие
      if (authResp) {
        localStorage.setItem('auth_session', 'true');
        setIsAuthenticated(true);
        setUser({ name: 'User' });
        return true;
      }
      return false;
    } catch (error) {
      console.error(error);
      return false;
    }
  };

  const logout = () => {
    localStorage.removeItem('auth_session');
    setIsAuthenticated(false);
    setUser(null);
  };

  return (
    <AuthContext.Provider
      value={{
        isAuthenticated,
        loginWithPassword,
        loginWithMnemonic,
        loginWithBiometrics,
        registerBiometrics,
        logout,
        user,
        biometricsSupported,
      }}
    >
      {children}
    </AuthContext.Provider>
  );
};

export const useAuth = () => {
  const ctx = useContext(AuthContext);
  if (!ctx) throw new Error('useAuth must be used within AuthProvider');
  return ctx;
};
```

⚠️ Важные замечания

1. Для реальной работы 1inch нужно указать правильные адреса токенов на Robinhood Chain. Пока используй тестовые, например, 0x... для USDC.
2. Для Jupiter – аналогично, нужны адреса токенов на Solana (например, EPjFWdd5AufqSSqeM2qN1xzybapC8G4wEGGkZwyTDt1v для USDC).
3. Биометрия в этом коде работает без бэкенда (данные хранятся в localStorage), но для продакшена обязателен сервер.
4. Токены Robinhood Chain можно получить через DexScreener API или через публичный RPC.
